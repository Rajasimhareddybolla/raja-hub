01 // BASIC & ADVANCED CRUD

1. Insert One

db.users.insertOne({ name: "Raja", age: 21 })
{ "acknowledged": true, "insertedId": "..." }

2. Insert Many

db.users.insertMany([{ name: "A" }, { name: "B" }])
{ "acknowledged": true, "insertedIds": [...] }

3. Find with Projection

db.users.find({}, { name: 1, _id: 0 })
{ "name": "Raja" }

4. Update with $set

db.users.updateOne({name: "Raja"}, {$set: {status: "Elite"}})
{ "matchedCount": 1, "modifiedCount": 1 }
02 // COMPARISON & LOGICAL

5. Greater Than ($gt)

db.users.find({ age: { $gt: 20 } })
[ { "name": "Raja", "age": 21 } ]

6. In Array ($in)

db.users.find({ status: { $in: ["Active", "Elite"] } })
[ { "name": "Raja", "status": "Elite" } ]

7. Logical AND

db.users.find({ $and: [{age: 21}, {status: "Elite"}] })
{ "name": "Raja" }

8. Logical OR

db.users.find({ $or: [{age: 21}, {name: "Admin"}] })
{ "name": "Raja" }
03 // ARRAY OPERATORS

9. Match All ($all)

db.users.find({ tags: { $all: ["ML", "AI"] } })
{ "tags": ["ML", "AI", "Cloud"] }

10. Element Match ($elemMatch)

db.users.find({ scores: { $elemMatch: { v: {$gt: 90} } } })
{ "name": "Raja", "scores": [{v: 95}] }

11. Array Size ($size)

db.users.find({ tags: { $size: 3 } })
{ "tags": ["A", "B", "C"] }

12. Push to Array ($push)

db.users.updateOne({name: "Raja"}, {$push: {tags: "Expert"}})
{ "modifiedCount": 1 }
04 // AGGREGATION PIPELINES

13. Match & Group

db.orders.aggregate([ {$match: {status: "A"}}, {$group: {_id: "$cust", total: {$sum: "$price"}}} ])
{ "_id": "Raja", "total": 5000 }

14. Unwind Array

db.users.aggregate([ {$unwind: "$tags"} ])
{ "tags": "ML" }, { "tags": "AI" } ...

15. Lookup (JOIN)

db.orders.aggregate([ {$lookup: {from: "inv", localField: "id", foreignField: "sku", as: "details"}} ])
{ "id": 1, "details": [...] }

16. Project & Rename

db.users.aggregate([ {$project: { userName: "$name", _id: 0 }} ])
{ "userName": "Raja" }
